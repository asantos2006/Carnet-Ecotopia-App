rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Función de ayuda: ¿Es el usuario el administrador?
    function esAdmin() {
      return request.auth != null && request.auth.token.email == 'ecotopia.asociacion@gmail.com';
    }

    // REGLAS PARA LA COLECCIÓN DE USUARIOS
    match /usuarios/{userId} {
      // 1. Cualquier usuario logueado puede crear su perfil (registro)
      allow create: if request.auth != null && request.auth.uid == userId;
      
      // 2. Un usuario solo puede leer sus propios datos, el Admin puede leer todos
      allow read: if request.auth != null && (request.auth.uid == userId || esAdmin());
      
      // 3. ¡CRÍTICO! Solo el Admin puede modificar puntos e historial
      // El usuario solo puede actualizar sus propios 'objetivosId' y 'metaPuntos'
      allow update: if esAdmin() || (
        request.auth.uid == userId && 
        !request.resource.data.diff(resource.data).affectedKeys().hasOnly(['puntosTotales', 'historial', 'email', 'nombre'])
      );
    }

    // REGLAS PARA LA COLECCIÓN DE EVENTOS
    match /eventos/{eventoId} {
      // 1. Todos los usuarios logueados pueden leer los eventos (para el panel de objetivos)
      allow read: if request.auth != null;
      
      // 2. Solo el Admin puede crear, borrar o editar eventos
      allow write: if esAdmin();
    }
  }
}
